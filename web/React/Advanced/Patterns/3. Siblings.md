# Siblings

A child component updates its parentâ€™s state, and the parent passes that state to a sibling component.

```jsx
// App.js
import React from 'react';
import logo from './logo.svg';
import './App.css';
import { Parent } from './Parent';

function App() {
  return (
    <div>
      <Parent />
    </div>
  );
}

export default App;
```

```jsx
// Parent.js
import React, { useState } from 'react';
import Child from './Child';
import Sibling from './sibling';

export function Parent() {
  const [value, setValue] = useState('guapa');

  const onchangeName = () => {
    setValue('guapa y bonica');
  };

  return (
    <div>
      <p>Cari es {value}</p>
      <Child onClick={onchangeName} />
      <Sibling value={value} />
    </div>
  );
}
```

```jsx
// Child.js

function Child({ onClick }) {
  const handleChange = () => {
    onClick();
  };

  return (
    <div>
      <button onClick={handleChange}>Press me!</button>
    </div>
  );
}

export default Child;
```

```jsx
// Sibling.js

function Sibling({ value }) {
  return <div>Argument: {value}</div>;
}

export default Sibling;
```
